fun bonusWhen(cargo: String): Float{
    var bonus = 0f
    when(cargo){
        "Gerente"->{ bonus = 2000f }
        "Coordenador"->{ bonus = 1500f }
        "Engenheiro de Software"->{ bonus = 1000f }
        "Estagiário"->{ bonus = 2000f }
        else->{0f}
    }
    return bonus
}

fun main() {
    println("Seu bônus é de: R$ ${bonusWhen("Gerente")}")
}
